JavaScript Implementation:
javascriptconst sendMessageButton = document.querySelector('.send-message-button');

const handleSendMessage = async () => {
  // 1. Show loading state
  sendMessageButton.classList.add('sending');
  sendMessageButton.textContent = 'Sending...';
  sendMessageButton.disabled = true;
  
  try {
    // 2. Send message (existing functionality)
    await sendMessage(); // Your existing function
    
    // 3. Show success state
    sendMessageButton.classList.remove('sending');
    sendMessageButton.classList.add('sent');
    sendMessageButton.textContent = 'Message Sent!';
    
    // 4. Reset button after 3 seconds (optional)
    setTimeout(() => {
      sendMessageButton.classList.remove('sent');
      sendMessageButton.textContent = 'Send Message Only';
      sendMessageButton.disabled = false;
    }, 3000);
    
  } catch (error) {
    // Handle error
    sendMessageButton.classList.remove('sending');
    sendMessageButton.textContent = 'Send Message Only';
    sendMessageButton.disabled = false;
    alert('Failed to send message. Please try again.');
  }
};
Recommended Approach: Combined Feedback
Show both button state change AND success message for best UX:
javascriptconst handleSendMessageOnly = async () => {
  const button = document.querySelector('.send-message-button');
  const successMessage = document.querySelector('.success-message');
  
  // 1. Disable button and show loading
  button.disabled = true;
  button.textContent = 'Sending...';
  button.style.opacity = '0.6';
  
  try {
    // 2. Execute existing send functionality
    await sendMessage(); // DO NOT CHANGE existing function
    
    // 3. Show success feedback
    button.textContent = 'Message Sent! ✓';
    button.style.backgroundColor = '#4CAF50';
    button.style.color = 'white';
    
    // 4. Show success message below button
    successMessage.textContent = 'Message Sent Successfully!';
    successMessage.style.display = 'block';
    
    // 5. Reset after 3 seconds
    setTimeout(() => {
      button.textContent = 'Send Message Only';
      button.style.backgroundColor = '';
      button.style.color = '';
      button.style.opacity = '1';
      button.disabled = false;
      successMessage.style.display = 'none';
    }, 3000);
    
  } catch (error) {
    // Reset on error
    button.textContent = 'Send Message Only';
    button.style.opacity = '1';
    button.disabled = false;
    alert('Failed to send message');
  }
};
