Full-Stack Queue Management System for Cafe 2020
Build a complete queue management system with QR code entry, real-time notifications, SMS integration, and time-based acceptance logic for efficient table allocation.

Technical Stack
Frontend

Framework: React with Vite
Styling: Tailwind CSS (core utility classes only)
UI Components: Shadcn UI
State Management: TanStack Query (React Query) for data fetching and real-time polling
Animations: Framer Motion for smooth transitions
Icons: Lucide React

Backend

Runtime: Node.js
Framework: Express.js
Database: MongoDB with Mongoose ODM
Authentication: JWT (JSON Web Tokens)
SMS Service: Twilio integration
Task Scheduling: node-cron for expiry checks

Twilio Setup
javascriptconst twilio = require('twilio');
const accountSid = process.env.TWILIO_ACCOUNT_SID;
const authToken = process.env.TWILIO_AUTH_TOKEN;
const client = twilio(accountSid, authToken);

const TWILIO_PHONE = process.env.TWILIO_PHONE_NUMBER;

async function sendSMS(phoneNumber, message, bookingId) {
  try {
    const formattedPhone = `+91${phoneNumber}`;
    
    const result = await client.messages.create({
      body: message,
      from: TWILIO_PHONE,
      to: formattedPhone
    });
    
    await Notification.create({
      bookingId,
      phoneNumber,
      message,
      status: 'sent',
      twilioSid: result.sid,
      sentAt: new Date()
    });
    
    await Queue.findByIdAndUpdate(bookingId, {
      notificationSent: true,
      notificationSentAt: new Date(),
      notificationStatus: 'sent',
      $inc: { notificationAttempts: 1 }
    });
    
    return { success: true, messageId: result.sid };
    
  } catch (error) {
    console.error('SMS failed:', error);
    
    await Notification.create({
      bookingId,
      phoneNumber,
      message,
      status: 'failed',
      error: error.message,
      attemptedAt: new Date()
    });
    
    return { success: false, error: error.message };
  }
}

// Retry logic with exponential backoff
async function sendWithRetry(phoneNumber, message, bookingId, maxAttempts = 3) {
  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    const result = await sendSMS(phoneNumber, message, bookingId);
    
    if (result.success) {
      return result;
    }
    
    if (attempt < maxAttempts) {
      await new Promise(resolve => setTimeout(resolve, 1000 * attempt));
    }
  }
  
  return { success: false, error: 'Max retry attempts reached' };
}

Auto-Expiry Function
javascriptasync function checkAndExpireBooking(bookingId) {
  const booking = await Queue.findById(bookingId);
  
  if (booking && booking.status === 'called' && booking.timerActive) {
    await Queue.findByIdAndUpdate(bookingId, {
      status: 'expired',
      timerActive: false,
      expiredAt: new Date(),
      responseType: 'expired'
    });
    
    const expirySMS = `Your booking at Cafe 2020 has expired due to no response. Please book again if interested.`;
    await sendSMS(booking.phoneNumber, expirySMS, bookingId);
    
    await callNextInQueue();
  }
}
Customer Acceptance Function
javascriptasync function acceptBooking(bookingId) {
  const booking = await Queue.findById(bookingId);
  
  if (booking.status !== 'called' || !booking.timerActive) {
    throw new Error('Booking is no longer valid for acceptance');
  }
  
  if (booking.expiryTimeoutId) {
    clearTimeout(booking.expiryTimeoutId);
  }
  
  await Queue.findByIdAndUpdate(bookingId, {
    status: 'confirmed',
    timerActive: false,
    respondedAt: new Date(),
    responseType: 'accepted'
  });
  
  const confirmSMS = `Thank you ${booking.name}! Your table #${booking.queueNumber} is confirmed at Cafe 2020. Please proceed to the counter.`;
  await sendSMS(booking.phoneNumber, confirmSMS, bookingId);
  
  return { success: true, message: 'Booking confirmed' };
}
Safety Cron Job (Backup Expiry Check)
javascriptconst cron = require('node-cron');

// Run every 1 minute
cron.schedule('* * * * *', async () => {
  const expiredBookings = await Queue.find({
    status: 'called',
    timerActive: true,
    responseDeadline: { $lt: new Date() }
  });
  
  for (const booking of expiredBookings) {
    await checkAndExpireBooking(booking._id);
  }
});
